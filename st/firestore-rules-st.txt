Pegar estos match dentro de tus reglas Firestore (junto a products, movements, users, etc.).

IMPORTANTE: hasta que NO publiques estas reglas en Firebase Console → Firestore → Reglas, el formulario
público verá «Missing or insufficient permissions» al enviar.

Formulario st/ingreso.html escribe SIN iniciar sesión (addDoc directo). El dash ST usa Google → request.auth != null.

NO uses signInAnonymously en ingreso si el proyecto tiene Anónimo deshabilitado (error auth/admin-restricted-operation).

    match /st_validaciones/{docId} {
      // Alta pública: solo create sin login, con validación de campos
      allow create: if request.resource.data.estado == 'pendiente'
        && request.resource.data.canal is string
        && request.resource.data.canal in ['P', 'E', 'S']
        && request.resource.data.nombre is string
        && request.resource.data.nombre.size() > 0
        && request.resource.data.correo is string
        && request.resource.data.correo.size() > 3
        && request.resource.data.producto is string
        && request.resource.data.producto.size() > 0
        && request.resource.data.modelo is string
        && request.resource.data.modelo.size() > 0
        && request.resource.data.falla1 is string
        && request.resource.data.obs is string
        && request.resource.data.obs.size() >= 10
        && request.resource.data.numero_seguimiento is string
        && request.resource.data.numero_seguimiento.matches('^[PES][0-9]{1,12}$');

      // Personal: leer / actualizar / borrar solo autenticado
      allow read, update, delete: if request.auth != null;
    }

    // Órdenes: informes (URLs) y técnico escriben aquí con sesión; Apps Script no usa Firestore del cliente para el Doc.
    match /st_ordenes/{docId} {
      allow read, write: if request.auth != null;
    }

    match /st_meta/{docId} {
      function isSeqDoc() {
        return docId == 'seq_P' || docId == 'seq_E' || docId == 'seq_S';
      }

      // seq_*: correlativo (ingreso público + aprobar). No afecta informes Docs/correo (solo st_ordenes + Web App).
      allow read: if isSeqDoc() || request.auth != null;

      allow create: if isSeqDoc()
        && request.resource.data.keys().hasOnly(['n'])
        && request.resource.data.n is int
        && request.resource.data.n >= 1
        && request.resource.data.n <= 200000;

      allow update: if isSeqDoc()
        && request.resource.data.n is int
        && resource.data.n is int
        && request.resource.data.n >= resource.data.n
        && request.resource.data.n <= resource.data.n + 50000
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['n']);

      allow delete: if false;

      // ordenes y demás meta: solo autenticados
      allow create, update, delete: if !isSeqDoc() && request.auth != null;
    }

    // Informes de salida por técnico (panel ST — implementación progresiva)
    match /st_salidas/{docId} {
      allow read: if request.auth != null;
      allow create, update: if request.auth != null;
      allow delete: if false;
    }

    // Log de envíos de correo (entrada/salida) — escritura desde cliente autenticado
    match /st_email_log/{docId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update, delete: if false;
    }

    // Perfiles ST (rol técnico / admin) — gestión recomendada con Admin SDK
    match /st_usuarios/{uid} {
      allow read: if request.auth != null && request.auth.uid == uid;
      allow write: if false;
    }
